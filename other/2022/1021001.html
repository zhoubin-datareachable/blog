    土豆 2022/10/6 windowscmd

    # 常用命令

    # echo

    打开或关闭命令回显功能,默认开启

    当echo关闭时，命令提示符不会出现在命令提示符窗口中。要再次显示命令提示符，请键入echo on。

    命令 描述
    echo on 开启命令回显
    echo off 关闭命令回显
    echo exp 打印要显示的文本
    echo. 打印空白行
    echo %var% 打印变量

    示例一

    # 开启命令回显
echo on
# 打印 hello world
echo hello world

    运行结果

    -->echo on
-->echo hello world
hello world

    示例二

    # 关闭命令回显
echo off
# 打印 hello world
echo hello world

    运行结果

    -->echo off
hello world

    示例三

    # 关闭命令回显
echo off
# 打印 hello world
echo hello
# 打印空行
echo.
# 打印 hello world
echo world

    运行结果

    -->echo off
hello
world

    示例四

    echo off
# 定义变量
set a=123
echo %a%

    运行结果

    123

    # setlocal

    setlocal 主要针对临时环境变量，不会影响到系统的变量环境设置，应与 endlocal 联用。

    # call

    在不停止父批处理程序的情况下从另一个调用一个批处理程序。该调用命令接受标签作为调用的目标

    命令 描述
    call :<label> 指定您希望批处理程序控件跳转到的标签。
    call :<label> <arguments> 指定要传递给批处理程序的新实例的命令行信息，从 开始:<label>。

    示例一

    echo off
# 调用 chnew
call :chnew
echo world
:chnew
echo hello

    运行结果

    hello
world
hello

    # goto

    此命令将命令处理定向到由标签标识的行。

    命令 描述
    goto :<label> 指定您希望批处理程序控件跳转到的标签。
    echo off
# 调用 chnew
goto :chnew
echo world
:chnew
echo hello

    运行结果

    hello

    # set

    此命令用于设置变量 示例一

    echo off
# 定义变量
set a=hello
set "b=world"
echo %a%
echo %b%

    运行结果

    hello
world

    示例二

    echo off
# 定义变量
set a=hello
# 引用变量a:
set c=%a:world%
echo %c%

    运行结果

    a:world

    示例三

    echo off
set a=hello world
echo %a%
# 字符串替换(目标字符串=替换后的字符串)
set b=%a:world=china%
echo %b%

    运行结果

    hello world
hello china

    示例四

    echo off
SETLOCAL
# 读取环境变量
echo %PATHEXT%
# 设置环境变量(去掉.JS)
set PATHEXT=%PATHEXT:;.JS;=;%
echo %PATHEXT%

    运行结果

    .COM;.EXE;.BAT;.CMD;.VBS;.VBE;.JS;.JSE;.WSF;.WSH;.MSC
.COM;.EXE;.BAT;.CMD;.VBS;.VBE;.JSE;.WSF;.WSH;.MSC

    示例五

    echo off
# 设置环境变量
set my_node=node
# 使用环境变量
%my_node% ./index.js
"%my_node%" "-v"

    运行结果

    index.js内容
v14.17.4

    示例六

    echo off
# ~dp0 当前路径
echo %~dp0
# 设置环境变量dp0
set dp0=%~dp0
node "%dp0%\index.js"

    运行结果

    C:\Users\Admin\Desktop\eslint_learn\cmd\
index.js内容

    示例七

    # 获取传递的参数 %*
echo off
echo %*

    运行结果

    -->echo off
hello

    # exit

    退出命令解释器或当前批处理脚本。

    命令 描述
    exit /b 退出当前批处理脚本。
    exit /b <exitcode> 推出代码

    示例一

    # 父进程
echo off
echo hello
call :find_dp0
echo world
# 结束父进程
exit /b
# 子进程
:find_dp0
set dp0=%~dp0
echo this is find_dp0
# 结束子进程
exit /b

    运行结果

    hello
this is find_dp0
world

    示例二

    echo off
echo?
# 错误等级
echo %errorlevel%

    运行结果

    'echo?' 不是内部或外部命令，也不是可运行的程序
或批处理文件。
9009

    示例三

    echo off
call :e1
echo %errorlevel%
exit /b
:e1
echo this is e1
exit /b 11

    运行结果

    this is e1
11

    # 实例npm脚本

    # 关闭命令回显
@ECHO off
# 开始设置临时环境变量
SETLOCAL
# 调用函数find_dp0
CALL :find_dp0

# 如果当前路径下存在node.exe
# 则设置_prog为当前node.exe
# 否则设置_prog为node(全局环境变量中的值)
# 追加可执行后缀.js
IF EXIST "%dp0%\node.exe" (
  SET "_prog=%dp0%\node.exe"
) ELSE (
  SET "_prog=node"
  SET PATHEXT=%PATHEXT:;.JS;=;%
)

# 执行命令
# %_prog% node所在路径
# %dp0% 当前CMD所在路径
"%_prog%"  "%dp0%\node_modules\npm\bin\npm-cli.js" %*
# 结束设置临时环境变量
ENDLOCAL
EXIT /b %errorlevel%

# 子进程 函数find_dp0
:find_dp0
# 设置变量dp0为当前路径
SET dp0=%~dp0
# 结束子进程
EXIT /b
